بخش اول: زمان‌بندی جلسه (۱۲۰ دقیقه)

این جلسه، آخرین جلسه دوره است. دانشجویان با مفاهیم پایه برنامه‌نویسی شیءگرا (OOP) آشنا می‌شوند و سپس کل دوره را مرور کرده و نقشه راه آینده را ترسیم می‌کنیم.

  1. مفهوم شیءگرایی (OOP) (۳۰ دقیقه): کلاس، شیء، ویژگی‌ها، متدها، سازنده (Constructor).
  2. ساخت کلاس Database با MySQLi (۳۰ دقیقه): کپسوله کردن اتصال و متدهای عمومی برای عملیات دیتابیس.
  3. بازنویسی پروژه با کلاس (۳۰ دقیقه): استفاده از کلاس Database به جای کدهای تکراری.
  4. جمع‌بندی و نقشه راه (۳۰ دقیقه): مرور کل دوره، معرفی MVC، لاراول و مسیر یادگیری آینده.
📌 نکته مهم برای مدرس:
  • هدف، آشنایی با مفاهیم پایه است تا دانشجویان برای یادگیری فریم‌ورک‌ها آماده شوند.
  • تأکید کنید که OOP یک تغییر نگرش در برنامه‌نویسی است.
  • جمع‌بندی و نقشه راه، انگیزه دانشجویان را برای ادامه مسیر حفظ می‌کند.

بخش دوم: مفاهیم پایه شیءگرایی (OOP) (۳۰ دقیقه)

🗣 کلمه به کلمه چه بگویید: «تا الان به‌صورت رویه‌ای (Procedural) کد می‌نوشتیم. یعنی توابع را می‌نوشتیم و آن‌ها را صدا می‌زدیم. برنامه‌نویسی شیءگرا (OOP) یک روش متفاوت است. در OOP، به جای توابع و متغیرهای جداگانه، همه چیز را در قالب اشیاء (Objects) سازماندهی می‌کنیم. یک کلاس (Class) مثل یک قالب یا نقشه است. یک شیء (Object) مثل یک نمونه ساخته شده از آن قالب است.»

۱. اجزای اصلی یک کلاس:

📋 کلاس (Class) → 📦 شیء (Object)
📝 ویژگی‌ها (Properties) + ⚡ متدها (Methods)

۲. ساختار یک کلاس در PHP:

<?php
// ===== تعریف یک کلاس ساده =====
class Car {
    // ===== ویژگی‌ها =====
    public $brand;
    public $color;
    public $speed = 0;
    
    // ===== سازنده =====
    public function __construct($brand, $color) {
        $this->brand = $brand;
        $this->color = $color;
    }
    
    // ===== متدها =====
    public function start() {
        echo "ماشین {$this->brand} روشن شد.<br>";
    }
    
    public function accelerate($amount) {
        $this->speed += $amount;
        echo "سرعت: {$this->speed}<br>";
    }
}

// ===== ایجاد شیء =====
$myCar = new Car("BMW", "آبی");
$myCar->start();
?>

۳. سطوح دسترسی (Visibility):

سطح دسترسی توضیح مثال
public قابل دسترس از همه جا public $name;
protected در کلاس و کلاس‌های فرزند protected $age;
private فقط در خود کلاس private $password;

۴. کلاس User با MySQLi:

<?php
// ===== کلاس User =====
class User {
    private $id;
    private $username;
    private $email;
    private $password;
    private $is_active;
    
    public function __construct($username, $email, $password) {
        $this->username = $username;
        $this->email = $email;
        $this->password = password_hash($password, PASSWORD_DEFAULT);
        $this->is_active = true;
    }
    
    public function getUsername() {
        return $this->username;
    }
    
    public function verifyPassword($password) {
        return password_verify($password, $this->password);
    }
}

// ===== استفاده =====
$user = new User("ali_dev", "ali@site.com", "123456");
echo "نام کاربری: " . $user->getUsername();
?>

بخش سوم: ساخت کلاس Database با MySQLi (۳۰ دقیقه)

🗣 کلمه به کلمه چه بگویید: «حالا می‌خواهیم کدهای اتصال به دیتابیس را به‌صورت یک کلاس بازنویسی کنیم. مزایای این کار: ۱. سازماندهی بهتر کدها. ۲. قابلیت استفاده مجدد در پروژه‌های دیگر. ۳. خوانایی بیشتر.»

کلاس Database با MySQLi:

<?php
// ===== فایل: classes/Database.php (MySQLi) =====

class Database {
    private $host;
    private $dbname;
    private $username;
    private $password;
    private $charset;
    private $conn;
    private $error;
    
    public function __construct($host = 'localhost', $dbname = 'guestbook_db', 
                                $username = 'root', $password = '', $charset = 'utf8mb4') {
        $this->host = $host;
        $this->dbname = $dbname;
        $this->username = $username;
        $this->password = $password;
        $this->charset = $charset;
        
        $this->connect();
    }
    
    private function connect() {
        $this->conn = new mysqli($this->host, $this->username, $this->password, $this->dbname);
        
        if ($this->conn->connect_error) {
            $this->error = $this->conn->connect_error;
            return false;
        }
        
        $this->conn->set_charset($this->charset);
        return true;
    }
    
    public function getConnection() {
        return $this->conn;
    }
    
    public function getError() {
        return $this->error;
    }
    
    public function query($sql, $params = []) {
        $stmt = $this->conn->prepare($sql);
        
        if ($params) {
            $types = '';
            $bind_params = [];
            foreach ($params as $key => $value) {
                if (is_int($value)) {
                    $types .= 'i';
                } elseif (is_double($value)) {
                    $types .= 'd';
                } elseif (is_string($value)) {
                    $types .= 's';
                } else {
                    $types .= 's';
                }
                $bind_params[] = $value;
            }
            $stmt->bind_param($types, ...$bind_params);
        }
        
        $stmt->execute();
        
        if (strpos(strtoupper($sql), 'SELECT') === 0) {
            $result = $stmt->get_result();
            $stmt->close();
            return $result;
        }
        
        $affected = $stmt->affected_rows;
        $insert_id = $stmt->insert_id;
        $stmt->close();
        
        return ['affected' => $affected, 'insert_id' => $insert_id];
    }
    
    public function fetchAll($sql, $params = []) {
        $result = $this->query($sql, $params);
        if ($result instanceof mysqli_result) {
            return $result->fetch_all(MYSQLI_ASSOC);
        }
        return false;
    }
    
    public function fetch($sql, $params = []) {
        $result = $this->query($sql, $params);
        if ($result instanceof mysqli_result) {
            return $result->fetch_assoc();
        }
        return false;
    }
    
    public function insert($table, $data) {
        $fields = array_keys($data);
        $placeholders = array_fill(0, count($fields), '?');
        
        $sql = "INSERT INTO {$table} (" . implode(', ', $fields) . ") 
                VALUES (" . implode(', ', $placeholders) . ")";
        
        $result = $this->query($sql, array_values($data));
        if ($result && isset($result['insert_id'])) {
            return $result['insert_id'];
        }
        return false;
    }
    
    public function update($table, $data, $where, $whereParams = []) {
        $set = [];
        foreach (array_keys($data) as $field) {
            $set[] = "{$field} = ?";
        }
        
        $sql = "UPDATE {$table} SET " . implode(', ', $set) . " WHERE {$where}";
        $params = array_merge(array_values($data), array_values($whereParams));
        
        $result = $this->query($sql, $params);
        if ($result && isset($result['affected'])) {
            return $result['affected'];
        }
        return false;
    }
    
    public function delete($table, $where, $params = []) {
        $sql = "DELETE FROM {$table} WHERE {$where}";
        $result = $this->query($sql, $params);
        if ($result && isset($result['affected'])) {
            return $result['affected'];
        }
        return false;
    }
    
    public function count($table, $where = '', $params = []) {
        $sql = "SELECT COUNT(*) as total FROM {$table}";
        if (!empty($where)) {
            $sql .= " WHERE {$where}";
        }
        $result = $this->fetch($sql, $params);
        return $result ? $result['total'] : 0;
    }
    
    public function close() {
        if ($this->conn) {
            $this->conn->close();
        }
    }
}
?>
